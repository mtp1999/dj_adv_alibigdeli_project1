    @method_decorator(cache_page(60 * 60))

# cache page for ViewSets
class MyViewSet(viewsets.ModelViewSet):
    ...

    @method_decorator(cache_page(60 * 60))
    def dispatch(self, request, *args, **kwargs):
        return super().dispatch(request, *args, **kwargs)
